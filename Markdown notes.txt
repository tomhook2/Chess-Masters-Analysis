```{r, echo = FALSE}
tblData <- readRDS("Chess Masters Data.rds")

PlayerName <- tblData %>%
  group_by(Master) %>%
  summarize(
    Count = n(),
    Percentage_TRUE = mean(Isf3PlayedMaster) * 100
  ) %>%
  filter(Count >= 100) %>%
  arrange(desc(Percentage_TRUE)) %>%
  slice(1) %>%
  pull(Master)
```
